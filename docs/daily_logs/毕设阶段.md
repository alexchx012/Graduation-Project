### A路线 - Linear Scalarization + Curriculum Weight Scheduling

#### 5.1 多目标Reward函数设计

```python
# 文件：mo_rewards.py

import torch

class MultiObjectiveRewards:
    """四个优化目标的reward函数"""
    
    def __init__(self, env):
        self.env = env
        
    def objective_1_velocity_tracking(self) -> torch.Tensor:
        """目标1：速度跟踪"""
        lin_vel = self.env.root_lin_vel_b[:, :2]
        cmd_lin_vel = self.env.commands[:, :2]
        
        lin_vel_error = torch.sum(torch.square(lin_vel - cmd_lin_vel), dim=1)
        reward = torch.exp(-5.0 * lin_vel_error)
        return reward
    
    def objective_2_energy_efficiency(self) -> torch.Tensor:
        """目标2：能量效率"""
        joint_torques = self.env.joint_torques
        torque_cost = torch.sum(torch.square(joint_torques), dim=1)
        reward = torch.exp(-1e-5 * torque_cost)
        return reward
    
    def objective_3_motion_smoothness(self) -> torch.Tensor:
        """目标3：动作平滑"""
        action_rate = torch.sum(
            torch.square(self.env.actions - self.env.last_actions), dim=1
        )
        reward = torch.exp(-0.01 * action_rate)
        return reward
    
    def objective_4_stability(self) -> torch.Tensor:
        """目标4：稳定性"""
        roll = self.env.root_euler[:, 0]
        pitch = self.env.root_euler[:, 1]
        orientation_penalty = torch.square(roll) + torch.square(pitch)
        reward = torch.exp(-2.0 * orientation_penalty)
        return reward
    
    def get_all_objectives(self) -> torch.Tensor:
        """返回所有目标的reward向量 [N, 4]"""
        r1 = self.objective_1_velocity_tracking()
        r2 = self.objective_2_energy_efficiency()
        r3 = self.objective_3_motion_smoothness()
        r4 = self.objective_4_stability()
        return torch.stack([r1, r2, r3, r4], dim=1)
```

#### 5.2 Linear Scalarization实现

```python
# 文件：linear_scalarization.py

class LinearScalarization:
    """线性加权多目标方法"""
    
    def __init__(self, weights: list = [0.25, 0.25, 0.25, 0.25]):
        self.weights = torch.tensor(weights)
        
    def scalarize(self, rewards: torch.Tensor) -> torch.Tensor:
        """将多目标reward加权求和
        Args:
            rewards: [N, 4] tensor
        Returns:
            [N] tensor of scalarized rewards
        """
        return torch.sum(rewards * self.weights, dim=1)
```

## 6. 多目标评价体系最终版

### 6.1 Pareto Front生成

```python
# pareto_analysis.py

import numpy as np
from pymoo.util.nds.non_dominated_sorting import NonDominatedSorting

def compute_pareto_front(objectives: np.ndarray) -> np.ndarray:
    """
    计算Pareto前沿
    Args:
        objectives: [N, 4] array, 每行是一个策略的4目标得分
                   注意：全部转为最大化问题（得分越高越好）
    Returns:
        pareto_front: Pareto前沿上的点
    """
    nds = NonDominatedSorting()
    fronts = nds.do(objectives, n_stop_if_ranked=len(objectives))
    pareto_indices = fronts[0]  # 第一前沿
    return objectives[pareto_indices]
```

### 6.2 Hypervolume计算

```python
from pymoo.indicators.hv import HV

def compute_hypervolume(pareto_front: np.ndarray, ref_point: np.ndarray) -> float:
    """
    计算Hypervolume指标
    Args:
        pareto_front: [M, 4] Pareto前沿上的点（已归一化到[0,1]）
        ref_point: [4] 参考点，通常取[0, 0, 0, 0]
    Returns:
        hypervolume: HV值，越大越好
    """
    # 注意：pymoo的HV计算假设最小化问题，需要取负
    ind = HV(ref_point=ref_point)
    hv = ind(pareto_front)
    return hv
```

### 6.3 归一化方法

```python
def normalize_objectives(objectives: np.ndarray) -> np.ndarray:
    """
    将目标归一化到[0, 1]范围
    使用所有策略的min-max归一化
    """
    min_vals = objectives.min(axis=0)
    max_vals = objectives.max(axis=0)
    normalized = (objectives - min_vals) / (max_vals - min_vals + 1e-8)
    return normalized
```

### 6.4 综合评分公式（答辩展示用）

```
综合评分 = 0.3×成功率 + 0.7×(w1×速度得分 + w2×能效得分 + w3×平滑得分 + w4×稳定得分)

其中：
- 成功率 = 1 - 跌倒次数/总步数
- 各目标得分为归一化后的值[0,1]
- w1-w4由用户偏好决定（默认均为0.25）
```
